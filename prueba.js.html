    /**
     * @constructor

    function Logger () {

    }

    /**
     * @alias message
     * @memberof! CompanyName.Logger#
     * @param {string} msg Message description
     * @param {string} [module] Module name
     * @param {object} [obj] Some object for logging
     */
    Logger.prototype.message = function (msg, module, obj) {
        if (obj)
            console.log("MSG:" + (module ? module + ":" : "") + msg, obj);
        else
            console.log("MSG:" + (module ? module + ":" : "") + msg);
    };

    ns.Logger = Logger;

})(CompanyName);
